function distro_specific() {
    if [[ ${ISO} = *".zip"* ]]; then
        unzip -qo "${VM_PATH}/${ISO}" -d "${VM_PATH}"
        rm -f "${VM_PATH}/${ISO}"
        ISO="$(ls -1 "${VM_PATH}/"*.iso)"
        ISO="$(basename "${ISO}")"
    fi
}

function specific_tweaks() {
    echo "boot=\"legacy\"" >> "${CONF_FILE}"
    echo "disk_size=\"12G\"" >> "${CONF_FILE}"
    echo "ram=\"2048M\"" >> "${CONF_FILE}"
}
